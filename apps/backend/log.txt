# Create the axes
axes = Axes([-1, 3range=[-1, 9_config={"color": BLUEshift(DOWN * 0.5)
# Create the graph of x^2
graph = axes.plot(lambda x: x**2, color=GREEN)
# Create the area under the curve
area = axes.get_area(graph, x_range=[0, 2], color=YELLOW, opacity=0.3)
# Create labels
x_label = axes.get_x_axis_label("x").shift(RIGHT * 3.5 + UP * 0.5)
y_label = axes.get_y_axis_label("y").shift(UP * 3.5 + LEFT * 0.5)
func_label = MathTex("f(x) = x^2").shift(UP * 3 + LEFT * 4)
integral_label = MathTex("\int x^2 dx = \frac{x^3}{3} + C").shift(UP * 3 + RIGHT * 3)
# Animate
self.play(Create(axes), Create(x_label), Create(y_label))
self.play(Create(graph))
self.play(Create(func_label))
self.play(Create(area))
self.play(Create(integral_label))
# Show the antiderivative
antideriv_graph = axes.plot(lambda x: (x**3)/3, color=RED)
self.play(Create(antideriv_graph))
# Clean up
self.play(FadeOut(area), FadeOut(func_label), FadeOut(integral_label))
final_label = MathTex("F(x) = \int f(x) dx = \frac{x^3}{3} + C").shift(UP * 3)
self.play(Create(final_label))
