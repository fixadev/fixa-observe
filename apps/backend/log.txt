# Create sun
sun = Circle(radius=0.5, fill_opacity=1, color=YELLOW)
self.play(Create(sun))
# Create planets
mercury = Dot(color=GRAY).shift(RIGHT * 1.5)
venus = Dot(color=ORANGE).shift(RIGHT * 2)
earth = Dot(color=BLUE).shift(RIGHT * 2.5)
mars = Dot(color=RED).shift(RIGHT * 3)
jupiter = Circle(radius=0.15, fill_opacity=1, color=ORANGE).shift(RIGHT * 3.7)
saturn = Circle(radius=0.13, fill_opacity=1, color=YELLOW).shift(RIGHT * 4.3)
uranus = Dot(color=BLUE).shift(RIGHT * 4.8)
neptune = Dot(color=BLUE).shift(RIGHT * 5.3)
self.play(Create(mercury), Create(venus), Create(earth), Create(mars), Create(jupiter), Create(saturn), Create(uranus), Create(neptune))
# Animate orbits
self.play(Rotate(mercury, angle=TAU*4, about_point=Rotate(venus, angle=TAU*3, about_point=Rotate(earth, angle=TAU*2.5, about_point=ORIGIN),, angle=TAU*2, about_point=ORIGIN),jupiter, angle=TAU*1.5, about_point=Rotate(saturn, angle=TAU, about_point=Rotate(uranus, angle=TAU*0.75, about_point=Rotate(neptune, angle=TAU*0.5, about_point=ORIGIN),=10,rate_func=linear
self.play(FadeOut(sun, mercury, venus, earth, mars, jupiter, saturn, uranus, neptune))
