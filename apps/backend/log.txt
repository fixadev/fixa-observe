# Create airplane shape
airplane = V().scale(0.5).rotate(-90 * DEGREES),(height=0.1, width=1.5).next_to(Triangle().scale(0.5).rotate(-90 * DEGREES), UP, buff=0)set_color(BLUE)
self.play(Create(airplane))
# Add ground
ground = Line(start=LEFT * 4, end=RIGHT * 4, color=GREEN).shift(DOWN * 3)
self.play(Create(ground))
# Show airflow
airflow_top = Arrow(start=LEFT * 3, end=RIGHT * 3, color=WHITE).shift(UP * 0.5)
airflow_bottom = Arrow(start=LEFT * 3, end=RIGHT * 3, color=WHITE).shift(DOWN * 0.5)
self.play(Create(airflow_top), Create(airflow_bottom))
# Explain lift
lift_arrow = Arrow(start=ORIGIN, end=UP, color=RED).next_to(airplane, RIGHT)
lift_text = Text("Lift", color=RED).next_to(lift_arrow, RIGHT)
self.play(Create(lift_arrow), Write(lift_text))
# Show pressure difference
pressure_high = Text("High Pressure", color=YELLOW).scale(0.5).next_to(airflow_bottom, DOWN)
pressure_low = Text("Low Pressure", color=BLUE).scale(0.5).next_to(airflow_top, UP)
self.play(Write(pressure_high), Write(pressure_low))
# Animate takeoff
self.play(airplane.animate.shift(UP * 2 + RIGHT * 2), rate_func=smooth)
self.play(FadeOut(ground), FadeOut(pressure_high), FadeOut(pressure_low), FadeOut(lift_arrow), FadeOut(lift_text), FadeOut(airflow_top), FadeOut(airflow_bottom))
# Show cruising
clouds = VGroup(0.3, color=WHITE).shift(LEFT * 3 + UP * 2radius=0.4, color=WHITE).shift(LEFT * 2.7 + UP * 2.2radius=0.3, color=WHITE).shift(LEFT * 2.5 + UP * 1.8)
)
self.play(Create(clouds))
self.play(airplane.animate.shift(RIGHT * 6), clouds.animate.shift(LEFT * 6))
self.play(FadeOut(airplane), FadeOut(clouds))
