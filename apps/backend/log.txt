title = Text("The Solar System").scale(0.8).shift(UP * 3)
self.play(Write(title))
sun = Circle(radius=0.5, fill_opacity=1, color=YELLOW)
mercury = Dot(color=GRAY)
venus = Dot(color=ORANGE)
earth = Dot(color=BLUE)
mars = Dot(color=RED)
jupiter = Circle(radius=0.2, fill_opacity=1, color=ORANGE)
saturn = Circle(radius=0.18, fill_opacity=1, color=YELLOW)
uranus = Dot(color=BLUE)
neptune = Dot(color=BLUE)
planets = VGroup(mercury, venus, earth, mars, jupiter, saturn, uranus, neptune)
orbits = VGroup(*[Circle(radius=i*0.5+1, color=WHITE, stroke_opacity=0.5) for i in range(8)])
self.play(FadeIn(sun), Create(orbits))
self.play(FadeIn(planets))
self.play(mercury, angle=TAU, about_point=ORIGIN, rate_func=linear(venus, angle=0.9*TAU, about_point=ORIGIN, rate_func=Rotate(earth, angle=0.8*TAU, about_point=ORIGIN, rate_func=Rotate(mars, angle=0.7*TAU, about_point=ORIGIN, rate_func=Rotate(jupiter, angle=0.4*TAU, about_point=ORIGIN, rate_func=Rotate(saturn, angle=0.3*TAU, about_point=ORIGIN, rate_func=linear(uranus, angle=0.2*TAU, about_point=ORIGIN, rate_func=linear(neptune, angle=0.1*TAU, about_point=ORIGIN, rate_func=linear),
self.play(FadeOut(title), FadeOut(sun), FadeOut(planets), FadeOut(orbits))
