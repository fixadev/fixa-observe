# Create sun
sun = Circle(radius=0.5, color=YELLOW, fill_opacity=1).shift(UP * 0.5)
self.play(Create(sun))
# Create planets
mercury = Dot(color=GRAY).shift(UP * 0.5 + RIGHT * 1.2)
venus = Dot(color=ORANGE).shift(UP * 0.5 + RIGHT * 1.8)
earth = Dot(color=BLUE).shift(UP * 0.5 + RIGHT * 2.4)
mars = Dot(color=RED).shift(UP * 0.5 + RIGHT * 3)
jupiter = Circle(radius=0.15, color=ORANGE, fill_opacity=1).shift(UP * 0.5 + RIGHT * 3.8)
saturn = Circle(radius=0.13, color=YELLOW, fill_opacity=1).shift(UP * 0.5 + RIGHT * 4.5)
uranus = Dot(color=BLUE).shift(UP * 0.5 + RIGHT * 5.1)
neptune = Dot(color=BLUE).shift(UP * 0.5 + RIGHT * 5.6)
self.play(Create(mercury), Create(venus), Create(earth), Create(mars), Create(jupiter), Create(saturn), Create(uranus), Create(neptune))
# Animate orbits
self.play(Rotate(mercury, angle=TAU * 4, about_point=sun.get_center(venus, angle=TAU * 3, about_point=sun.get_center(earth, angle=TAU * 2, about_point=sun.get_center(mars, angle=TAU * 1.5, about_point=sun.get_Rotate(jupiter, angle=TAU, about_point=sun.get_center()),, angle=TAU * 0.8, about_point=sun.get_center(uranus, angle=TAU * 0.6, about_point=sun.get_center(neptune, angle=TAU * 0.4, about_point=sun.get_center()),run_time=8
)
self.play(FadeOut(sun, mercury, venus, earth, mars, jupiter, saturn, uranus, neptune))
